PDF Loader PDF
Three approaches — use PyMuPDF for speed, PDFPlumber for tables, Unstructured for complex layouts.
1
Load the PDF file

Use a LangChain loader — it wraps the underlying library and returns Document objects

2
Each page → one Document

page_content = extracted text, metadata = filename, page number, source

3
Inspect the output

Print page_content and metadata to understand what you got

Method A — PyMuPDF (fastest, recommended)

pdf_loader_pymupdf.py
from langchain_community.document_loaders import PyMuPDFLoader

# --- Load a single PDF ---
loader = PyMuPDFLoader("data/my_document.pdf")
docs = loader.load()   # returns list of Document objects

# --- Inspect the output ---
print(f"Total pages loaded: {len(docs)}")
print(f"\n--- Page 1 content (first 300 chars) ---")
print(docs[0].page_content[:300])
print(f"\n--- Metadata ---")
print(docs[0].metadata)

# Example output:
# Total pages loaded: 12
# --- Page 1 content ---
# "Introduction to Machine Learning..."
# --- Metadata ---
# {'source': 'data/my_document.pdf', 'page': 0, 'total_pages': 12}

# --- Load ALL PDFs from a folder ---
import os
from pathlib import Path

all_docs = []
pdf_folder = Path("data/pdfs/")

for pdf_file in pdf_folder.glob("*.pdf"):
    loader = PyMuPDFLoader(str(pdf_file))
    docs = loader.load()
    all_docs.extend(docs)
    print(f"Loaded {len(docs)} pages from {pdf_file.name}")

print(f"\nTotal documents across all PDFs: {len(all_docs)}")

Method B — PDFPlumber (best for tables inside PDFs)

pdf_loader_plumber.py
from langchain_community.document_loaders import PDFPlumberLoader
import pdfplumber

# --- Via LangChain wrapper ---
loader = PDFPlumberLoader("data/report_with_tables.pdf")
docs = loader.load()

# --- Direct pdfplumber for table extraction ---
with pdfplumber.open("data/report_with_tables.pdf") as pdf:
    for i, page in enumerate(pdf.pages):
        text = page.extract_text()
        tables = page.extract_tables()  # list of tables per page

        print(f"\n=== Page {i+1} ===")
        print(f"Text: {text[:200] if text else 'none'}")
        print(f"Tables found: {len(tables)}")

        for j, table in enumerate(tables):
            print(f"  Table {j+1}: {len(table)} rows x {len(table[0])} cols")
            print(f"  Header: {table[0]}")   # first row = headers
            print(f"  Row 2 : {table[1]}")   # second row = first data
Best practice: Use PyMuPDF for text-heavy PDFs (research papers, manuals). Switch to PDFPlumber when you know the PDF contains structured tables. You can combine both in the same pipeline.

Method C — DirectoryLoader (load entire folder automatically)

directory_loader.py
from langchain_community.document_loaders import DirectoryLoader, PyMuPDFLoader

loader = DirectoryLoader(
    path="data/",           # folder to scan

Image Loader + OCR IMAGE
Extract text from images using Tesseract OCR, and convert scanned PDFs to text.
Tesseract must be installed system-wide before pytesseract will work. See the Setup tab for OS-specific instructions.
1
Load image with Pillow

Open PNG/JPG files, optionally preprocess (grayscale, contrast) for better OCR

2
Run Tesseract OCR

pytesseract.image_to_string() returns raw text from the image

3
Wrap in Document object

Create LangChain Document manually so it fits into the rest of the pipeline

Method A — OCR from image files

image_ocr_loader.py
import pytesseract
from PIL import Image, ImageEnhance, ImageFilter
from langchain.schema import Document
from pathlib import Path

def preprocess_image(img):
    """Improve OCR accuracy by preprocessing the image."""
    img = img.convert('L')              # to grayscale
    img = ImageEnhance.Contrast(img).enhance(2.0)  # boost contrast
    img = img.filter(ImageFilter.SHARPEN)  # sharpen edges
    return img

def load_image_as_document(image_path: str) -> Document:
    img = Image.open(image_path)
    img = preprocess_image(img)

    # Run OCR — lang='eng' for English, 'hin' for Hindi, etc.
    text = pytesseract.image_to_string(img, lang='eng')
    text = text.strip()

    return Document(
        page_content=text,
        metadata={
            "source": image_path,
            "type": "image",
            "filename": Path(image_path).name
        }
    )

# --- Load all images from a folder ---
image_folder = Path("data/images/")
image_docs = []

for ext in ["*.png", "*.jpg", "*.jpeg", "*.tiff"]:
    for img_path in image_folder.glob(ext):
        doc = load_image_as_document(str(img_path))
        if doc.page_content:       # skip blank/empty images
            image_docs.append(doc)
            print(f"✓ {img_path.name}: {len(doc.page_content)} chars extracted")
        else:
            print(f"✗ {img_path.name}: no text found (maybe a photo?)")

print(f"\nTotal image documents: {len(image_docs)}")

Method B — Scanned PDF → Images → OCR

scanned_pdf_ocr.py
from pdf2image import convert_from_path
import pytesseract
from langchain.schema import Document

def load_scanned_pdf(pdf_path: str) -> list[Document]:
    """Convert each page of a scanned PDF to an image, then OCR it."""
    # dpi=300 is standard for good OCR quality
    pages = convert_from_path(pdf_path, dpi=300)
    docs = []

    for i, page_image in enumerate(pages):
        text = pytesseract.image_to_string(page_image, lang='eng')
        docs.append(Document(
            page_content=text.strip(),
            metadata={
                "source": pdf_path,
                "page": i,
                "type": "scanned_pdf"
            }
        ))
        print(f"  Page {i+1}: {len(text)} chars")

    return docs

scanned_docs = load_scanned_pdf("data/scanned_report.pdf")
print(f"Extracted {len(scanned_docs)} pages from scanned PDF")
Study tip: A scanned PDF has no embedded text — it's just images of pages. PyMuPDFLoader will return empty strings for these. Always check doc.page_content.strip() is not empty after loading PDFs — if it is, you have a scanned PDF and need OCR.

Method B — pandas for full control

pandas_loader.py
import pandas as pd
from langchain.schema import Document

def load_csv_as_documents(csv_path: str, text_columns: list) -> list[Document]:
    """Load CSV where specific columns are combined as page_content."""
    df = pd.read_csv(csv_path)
    docs = []

    for idx, row in df.iterrows():
        # Combine selected columns into readable text
        content = "\n".join([
            f"{col}: {row[col]}"
            for col in text_columns
            if pd.notna(row[col])
        ])
        docs.append(Document(
            page_content=content,
            metadata={
                "source": csv_path,
                "row": idx,
                "type": "csv"
            }
        ))
    return docs

# --- Excel: same approach ---
def load_excel_as_documents(excel_path: str, sheet_name=0) -> list[Document]:
    df = pd.read_excel(excel_path, sheet_name=sheet_name)
    docs = []
    for idx, row in df.iterrows():
        content = row.to_string()   # all columns as text
        docs.append(Document(
            page_content=content,
            metadata={"source": excel_path, "row": idx, "sheet": sheet_name}
        ))
    return docs

# Usage:
csv_docs = load_csv_as_documents(
    "data/employees.csv",
    text_columns=["name", "role", "department", "skills"]
)
print(f"Loaded {len(csv_docs)} row documents")
print(csv_docs[0].page_content)
Study tip: For numeric data (sales figures, measurements), convert the whole table to a markdown string — this preserves relationships between rows/columns and makes it easier for the LLM to reason about the data.

JSON files

json_loader.py
from langchain_community.document_loaders import JSONLoader
import json
from langchain.schema import Document

# --- JSONLoader with jq schema to pick fields ---
# Example JSON: [{"id":1, "title":"...", "body":"...", "tags":["a","b"]}]
loader = JSONLoader(
    file_path="data/articles.json",
    jq_schema=".[] | .body",       # extract 'body' from each item
    text_content=True
)
json_docs = loader.load()

# --- Manual JSON load for full control ---
def load_json_as_documents(json_path: str) -> list[Document]:
    with open(json_path) as f:
        data = json.load(f)

    docs = []
    for item in data:
        content = f"""
Title: {item.get('title', '')}
Content: {item.get('body', '')}
Tags: {', '.join(item.get('tags', []))}
        """.strip()

        docs.append(Document(
            page_content=content,
            metadata={
                "source": json_path,
                "id": item.get("id"),
                "type": "json"
            }
        ))
    return docs
Full Ingestion Pipeline
One script that loads all data types, tags them, and produces a unified list of Documents ready for chunking.
This is your practice file. Run this against a folder of mixed documents and inspect the output. The next stage (chunking) takes this all_docs list as input.
stage1_ingestion_pipeline.py
"""
RAG Pipeline — Stage 1: Data Ingestion
Handles: PDF, Scanned PDF (OCR), Images, CSV, Excel, TXT, MD, HTML, JSON
Output : list[Document] ready for Stage 2 (chunking)
"""

import os
from pathlib import Path
import pandas as pd
import pytesseract
from PIL import Image

from langchain.schema import Document
from langchain_community.document_loaders import (
    PyMuPDFLoader,
    CSVLoader,
    TextLoader,
    BSHTMLLoader,
    JSONLoader,
    UnstructuredMarkdownLoader,
)


# ─────────────────────────────────────────────
# 1. PDF LOADER
# ─────────────────────────────────────────────
def ingest_pdfs(folder: str) -> list[Document]:
    docs = []
    for pdf in Path(folder).glob("*.pdf"):
        loader = PyMuPDFLoader(str(pdf))
        pages = loader.load()
        for doc in pages:
            doc.metadata["data_type"] = "pdf"  # tag the type
        docs.extend(pages)
        print(f"  [PDF] {pdf.name} → {len(pages)} pages")
    return docs


# ─────────────────────────────────────────────
# 2. IMAGE OCR LOADER
# ─────────────────────────────────────────────
def ingest_images(folder: str) -> list[Document]:
    docs = []
    extensions = ["*.png", "*.jpg", "*.jpeg", "*.tiff"]
    for ext in extensions:
        for img_path in Path(folder).glob(ext):
            try:
                img = Image.open(img_path).convert("L")
                text = pytesseract.image_to_string(img).strip()
                if text:
                    docs.append(Document(
                        page_content=text,
                        metadata={
                            "source": str(img_path),
                            "data_type": "image",
                            "filename": img_path.name
                        }
                    ))
                    print(f"  [IMG] {img_path.name} → {len(text)} chars")
            except Exception as e:
                print(f"  [IMG] {img_path.name} → ERROR: {e}")
    return docs


# ─────────────────────────────────────────────
# 3. CSV / EXCEL LOADER
# ─────────────────────────────────────────────
def ingest_tables(folder: str) -> list[Document]:
    docs = []
    # CSV
    for csv_path in Path(folder).glob("*.csv"):
        df = pd.read_csv(csv_path)
        for idx, row in df.iterrows():
            docs.append(Document(
                page_content=row.to_string(),
                metadata={"source": str(csv_path), "row": idx, "data_type": "csv"}
            ))
        print(f"  [CSV] {csv_path.name} → {len(df)} rows")
    # Excel
    for xl_path in Path(folder).glob("*.xlsx"):
        df = pd.read_excel(xl_path)
        for idx, row in df.iterrows():
            docs.append(Document(
                page_content=row.to_string(),
                metadata={"source": str(xl_path), "row": idx, "data_type": "excel"}
            ))
        print(f"  [XLS] {xl_path.name} → {len(df)} rows")
    return docs


# ─────────────────────────────────────────────
# 4. TEXT / MD / HTML / JSON LOADER
# ─────────────────────────────────────────────
def ingest_text_files(folder: str) -> list[Document]:
    docs = []
    loaders = {
        "*.txt":  (TextLoader,                  "txt"),
        "*.md":   (UnstructuredMarkdownLoader,   "markdown"),
        "*.html": (BSHTMLLoader,               "html"),
    }
    for pattern, (loader_cls, dtype) in loaders.items():
        for fpath in Path(folder).glob(pattern):
            try:
                loaded = loader_cls(str(fpath)).load()
                for d in loaded:
                    d.metadata["data_type"] = dtype
                docs.extend(loaded)
                print(f"  [{dtype.upper()}] {fpath.name} → {len(loaded)} docs")
            except Exception as e:
                print(f"  ERROR loading {fpath.name}: {e}")
    return docs


# ─────────────────────────────────────────────
# MAIN — RUN ALL LOADERS
# ─────────────────────────────────────────────
def run_ingestion(data_folder: str) -> list[Document]:
    print(f"\n{'='*50}")
    print(f"RAG Stage 1 — Ingesting from: {data_folder}")
    print(f"{'='*50}\n")

    all_docs = []
    all_docs.extend(ingest_pdfs(data_folder))
    all_docs.extend(ingest_images(data_folder))
    all_docs.extend(ingest_tables(data_folder))
    all_docs.extend(ingest_text_files(data_folder))

    print(f"\n{'='*50}")
    print(f"✅ Total documents ingested: {len(all_docs)}")

    # Summary by type
    from collections import Counter
    counts = Counter(d.metadata.get("data_type") for d in all_docs)
    for dtype, count in counts.items():
        print(f"  {dtype:10s}: {count} documents")
    print(f"{'='*50}\n")

    return all_docs   # ← pass this to Stage 2 (chunking)


if __name__ == "__main__":
    docs = run_ingestion("./data")

    # Quick sanity check
    print("Sample document preview:")
    print("  Source:", docs[0].metadata['source'])
    print("  Type  :", docs[0].metadata['data_type'])
    print("  Text  :", docs[0].page_content[:200])
How to practice: Create a data/ folder, drop in one PDF, one image (screenshot), one CSV, and one .txt file. Run this script. When you see the summary output with counts per type, Stage 1 is working. That all_docs list flows directly into Stage 2.
