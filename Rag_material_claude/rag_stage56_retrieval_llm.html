The golden rule of RAG: Retrieval quality > LLM quality. A powerful LLM can't save bad retrieval. Invest 80% of your effort in stages 1–5. A well-retrieved set of 5 chunks with a medium LLM beats poorly-retrieved chunks with GPT-4o.

① Basic Similarity Search

retrieval_basic.py
# vectorstore = your built vector store from Stage 4

# ── Top-k similarity search ────────────────────────────
query = "What are the side effects of aspirin?"
results = vectorstore.similarity_search(query, k=5)
for i, doc in enumerate(results):
    print(f"[{i+1}] {doc.metadata.get('source')} | {doc.page_content[:120]}")

# ── With relevance scores ──────────────────────────────
results = vectorstore.similarity_search_with_relevance_scores(query, k=5)
for doc, score in results:
    print(f"Score: {score:.4f} | {doc.page_content[:100]}")

# ── Score threshold — only return if above cutoff ─────
retriever = vectorstore.as_retriever(
    search_type="similarity_score_threshold",
    search_kwargs={"score_threshold": 0.75, "k": 5}
)
# Only returns chunks with similarity ≥ 0.75
# Prevents returning irrelevant chunks when query is out-of-domain
docs = retriever.invoke(query)
print(f"Relevant docs above threshold: {len(docs)}")

② MMR — Maximum Marginal Relevance (Diversity-aware retrieval)

Problem MMR solves: Without MMR, top-5 results are often 5 near-identical chunks that all say the same thing — wasting your context window. MMR balances relevance + diversity to return broader coverage.
retrieval_mmr.py
# MMR: fetch_k candidates, then select k most diverse among them
retriever_mmr = vectorstore.as_retriever(
    search_type="mmr",
    search_kwargs={
        "k": 5,          # final number to return
        "fetch_k": 20,   # candidates to fetch before diversity filtering
        "lambda_mult": 0.5  # 0=max diversity, 1=max relevance, 0.5=balanced
    }
)
docs = retriever_mmr.invoke("What is RAG?")
print(f"MMR returned {len(docs)} diverse chunks")

# When to use MMR:
# - Long documents covering many sub-topics
# - When you want broad coverage of a topic
# - When top-k returns near-duplicate chunks

    retrievers=[bm25_retriever, dense_retriever],
    weights=[0.4, 0.6]   # 40% BM25 + 60% semantic
)

# When to use ensemble:
# - Queries with specific technical terms (BM25 catches exact keywords)
# - When semantic search alone misses exact acronyms/names
# - Best of both worlds for most production RAG systems

① Multi-Query Retriever — Generate multiple queries automatically

Problem it solves: A single query may miss relevant chunks depending on exact phrasing. Multi-query generates 3–5 different phrasings of your question and retrieves for all of them — dramatically improving recall.
retrieval_multiquery.py
from langchain.retrievers.multi_query import MultiQueryRetriever
from langchain_community.llms import Ollama

# Uses LLM to rewrite your question 5 different ways
# then retrieves for ALL of them and deduplicates results
llm = Ollama(model="llama3.2")   # any LLM works here

mq_retriever = MultiQueryRetriever.from_llm(
    retriever=vectorstore.as_retriever(search_kwargs={"k": 4}),
    llm=llm,
    include_original=True  # also include original query results
)

# The LLM generates variants like:
# Original: "What is RAG?"
# Variant 1: "How does retrieval augmented generation work?"
# Variant 2: "Explain the concept of RAG in NLP"
# Variant 3: "What is the purpose of retrieval in language models?"
# Retrieves for ALL → deduplicates → returns unique set

docs = mq_retriever.invoke("What is RAG?")
print(f"Multi-query returned {len(docs)} unique chunks")

② HyDE — Hypothetical Document Embeddings

Brilliant trick: Instead of embedding the question directly, ask the LLM to write a hypothetical answer first, then embed that answer. A hypothetical answer is more similar to real document chunks than the question itself.
retrieval_hyde.py
from langchain.prompts import PromptTemplate
from langchain_community.llms import Ollama
from langchain_huggingface import HuggingFaceEmbeddings
import numpy as np

llm = Ollama(model="llama3.2")
embeddings = HuggingFaceEmbeddings(model_name="BAAI/bge-small-en-v1.5")

def hyde_retrieve(query: str, vectorstore, k: int = 5):
    """HyDE: generate hypothetical answer, embed it, retrieve similar chunks."""

    # Step 1: Generate hypothetical answer
    hyde_prompt = f"""Write a detailed, factual paragraph that would answer 
this question. Write as if from a technical document. Do not mention 
you are hypothesizing — just write the answer directly.

Question: {query}
Answer:"""
    hypo_answer = llm.invoke(hyde_prompt)
    print(f"Hypothetical answer: {hypo_answer[:100]}...")

    # Step 2: Embed the hypothetical answer (not the question!)
    hypo_embedding = embeddings.embed_query(hypo_answer)

    # Step 3: Search with hypothetical embedding
    docs = vectorstore.similarity_search_by_vector(hypo_embedding, k=k)
    return docs

results = hyde_retrieve("How does HNSW indexing work?", vectorstore)
print(f"HyDE retrieved: {len(results)} chunks")

③ Contextual Compression — Filter irrelevant content within chunks

retrieval_compression.py
from langchain.retrievers import ContextualCompressionRetriever
from langchain.retrievers.document_compressors import LLMChainExtractor
from langchain_community.llms import Ollama

llm = Ollama(model="llama3.2")

# Compressor: given chunk + query → extract only the relevant sentences
compressor = LLMChainExtractor.from_llm(llm)

# Wrap your base retriever with compression
compression_retriever = ContextualCompressionRetriever(
    base_compressor=compressor,
    base_retriever=vectorstore.as_retriever(search_kwargs={"k": 5})
)

# Now returned documents contain ONLY the relevant sentences from each chunk
# — not the whole 1000-char chunk
docs = compression_retriever.invoke("What is the chunk overlap parameter?")
for doc in docs:
    print(f"Compressed chunk ({len(doc.page_content)} chars):\n{doc.page_content}\n")

④ Reranking — Cross-encoder for precision

retrieval_rerank.py — pip install sentence-transformers
from sentence_transformers import CrossEncoder
from langchain.schema import Document

# Cross-encoder: reads query + chunk TOGETHER — much more accurate than bi-encoder
# Use: fetch top-20 with fast ANN, rerank with cross-encoder, take top-5
cross_encoder = CrossEncoder("cross-encoder/ms-marco-MiniLM-L-6-v2")

def rerank_documents(query: str, docs: list[Document], top_k: int = 5):
    """Rerank retrieved docs using a cross-encoder for higher precision."""
    pairs = [(query, doc.page_content) for doc in docs]
    scores = cross_encoder.predict(pairs)

    ranked = sorted(zip(scores, docs), key=lambda x: x[0], reverse=True)
    return [doc for _, doc in ranked[:top_k]]

# Two-stage retrieval pipeline
query = "How does HNSW graph search work?"
# Stage 1: Fast ANN fetch (top-20, less precise)
candidates = vectorstore.similarity_search(query, k=20)
# Stage 2: Cross-encoder rerank (top-5, highly precise)
final_docs = rerank_documents(query, candidates, top_k=5)

print(f"After reranking: {len(final_docs)} precision-ranked chunks")
for i, doc in enumerate(final_docs):
    print(f"[{i+1}] {doc.page_content[:120]}")

⑤ Parent Document Retriever — Index child, return parent

retrieval_parent_document.py
from langchain.retrievers import ParentDocumentRetriever
from langchain.storage import InMemoryStore
from langchain_text_splitters import RecursiveCharacterTextSplitter

# Child: small chunks (for precise embedding + retrieval)
# Parent: large chunks (for rich LLM context)
# Index child chunks → retrieve → return parent chunk

child_splitter = RecursiveCharacterTextSplitter(chunk_size=200)  # small
parent_splitter = RecursiveCharacterTextSplitter(chunk_size=1000) # large

store = InMemoryStore()  # stores parent docs

retriever = ParentDocumentRetriever(
    vectorstore=vectorstore,
    docstore=store,
    child_splitter=child_splitter,
    parent_splitter=parent_splitter
)
retriever.add_documents(stage1_docs)

# Query matches small child chunks (precise) but returns large parent chunks (rich)
docs = retriever.invoke("What is chunk overlap?")
print(f"Returned {len(docs)} parent chunks")
print(f"Parent chunk size: {len(docs[0].page_content)} chars")

ollama_rag.py — full RAG with 100% local models
from langchain_ollama import OllamaLLM, OllamaEmbeddings, ChatOllama
from langchain_chroma import Chroma
from langchain.prompts import ChatPromptTemplate
from langchain.schema.runnable import RunnablePassthrough
from langchain.schema.output_parser import StrOutputParser

# ── 100% LOCAL — no internet needed after model download ──

# Local embedding model
embeddings = OllamaEmbeddings(model="nomic-embed-text")

# Load vector store
vectorstore = Chroma(
    persist_directory="./chroma_db",
    embedding_function=embeddings
)
retriever = vectorstore.as_retriever(search_kwargs={"k": 5})

# Local LLM
llm = ChatOllama(
    model="llama3.2",
    temperature=0,          # 0 = deterministic, good for RAG
    num_ctx=4096,           # context window for the LLM
)

# RAG prompt template
rag_prompt = ChatPromptTemplate.from_template("""
You are a helpful AI assistant. Answer the question using ONLY 
the context provided below. If the answer is not in the context, 
say "I don't have enough information to answer this."

Context:
{context}

Question: {question}

Answer:""")

# Format retrieved docs into a string
def format_docs(docs):
    return "\n\n---\n\n".join([
        f"Source: {d.metadata.get('source','unknown')}\n{d.page_content}"
        for d in docs
    ])

# Build LCEL chain (LangChain Expression Language)
rag_chain = (
    {"context": retriever | format_docs, "question": RunnablePassthrough()}
    | rag_prompt
    | llm
    | StrOutputParser()
)

# Run the chain
question = "What is the difference between chunking and embedding?"
answer = rag_chain.invoke(question)
print(f"Q: {question}")
print(f"A: {answer}")

# ── Streaming output (see tokens as they generate) ────
print("\n--- Streaming ---")
for chunk in rag_chain.stream(question):
    print(chunk, end="", flush=True)
print()

Prompt Engineering for RAG CRITICAL SKILL
The prompt template is what transforms raw retrieved chunks into a grounded, accurate, useful answer. These templates are battle-tested for production RAG.
The 3 components of every RAG prompt:
System: Role + rules (stay grounded, don't hallucinate, cite sources)
Context: Retrieved chunks from your vector store
Question: The user's actual question

Template 1 — Basic RAG Prompt

SYSTEM
You are a helpful assistant. Answer questions ONLY using the context below. If the answer is not in the context, say exactly: "I don't have enough information." Do not make up facts. Be concise and accurate.
CONTEXT (retrieved chunks)
{context}
QUESTION
{question}

Template 2 — Citation RAG Prompt (production-grade)

SYSTEM
You are a precise document assistant. Rules: 1. Answer using ONLY the provided context. Never add outside knowledge. 2. After your answer, list sources as [Source: filename, page N]. 3. If context is insufficient, say: "The provided documents don't contain this information." 4. Keep answers concise — 2-4 sentences unless detail is essential.
CONTEXT
[1] Source: rag_guide.pdf, Page 3 Retrieval Augmented Generation combines a retrieval system with a language model... [2] Source: embedding_tutorial.md, Section: Basics Embeddings are dense vector representations of text...
QUESTION
How do embeddings work in a RAG system?

# ── Format context with source labels ────────────────
def format_docs_with_sources(docs):
    """Format retrieved docs with numbered source labels for citation."""
    formatted = []
    for i, doc in enumerate(docs, 1):
        source = doc.metadata.get('source', 'unknown')
        page   = doc.metadata.get('page', '')
        label  = f"[{i}] Source: {source}" + (f", Page {page}" if page else "")
        formatted.append(f"{label}\n{doc.page_content}")
    return "\n\n".join(formatted)

Pattern 1 — Simple RAG Chain (LCEL)

chain_simple.py
from langchain.schema.runnable import RunnablePassthrough, RunnableParallel
from langchain.schema.output_parser import StrOutputParser
from langchain_ollama import ChatOllama
from langchain.prompts import ChatPromptTemplate

llm = ChatOllama(model="llama3.2", temperature=0)
retriever = vectorstore.as_retriever(search_kwargs={"k": 5})

prompt = ChatPromptTemplate.from_messages([
    ("system", "Answer using ONLY this context. If not in context, say so.\n\nContext:\n{context}"),
    ("human", "{question}")
])

# LCEL chain — reads left to right
rag_chain = (
    RunnableParallel({
        "context":  retriever | format_docs_with_sources,
        "question": RunnablePassthrough()
    })
    | prompt
    | llm
    | StrOutputParser()
)

answer = rag_chain.invoke("What is semantic chunking?")
print(answer)

# ── Retrieval metrics ─────────────────────────────────
def measure_retrieval(queries: list[str], relevant_docs: dict, k=5):
    """Hit rate: how often does top-k contain the relevant doc?"""
    hits = 0
    for query in queries:
        retrieved = [d.metadata['source'] for d in retriever.invoke(query)]
        expected = relevant_docs.get(query, [])
        if any(e in retrieved for e in expected):
            hits += 1
    print(f"Hit Rate @{k}: {hits/len(queries):.1%} ({hits}/{len(queries)})")

rag_pipeline_complete.py — ALL 6 STAGES, production-ready
"""
COMPLETE RAG PIPELINE — All 6 Stages
=====================================
Stage 1: Ingest documents (PDF, images, CSV, text, HTML)
Stage 2: Chunk documents (recursive, markdown, sentence-aware)
Stage 3: Embed chunks (local HuggingFace or Ollama)
Stage 4: Vector store (ChromaDB local, persistent)
Stage 5: Retrieve relevant chunks (MMR + metadata filter)
Stage 6: Generate answer (local Ollama LLM)

Requirements:
  pip install langchain langchain-community langchain-chroma
              langchain-huggingface langchain-ollama
              pymupdf pdfplumber pytesseract pillow
              pandas sentence-transformers chromadb

  ollama pull llama3.2
  ollama pull nomic-embed-text
"""

import os, sys
from pathlib import Path
from langchain.schema import Document
from langchain.prompts import ChatPromptTemplate
from langchain.schema.runnable import RunnablePassthrough, RunnableParallel
from langchain.schema.output_parser import StrOutputParser

# ══════════════════════════════════════════════════════
# CONFIGURATION
# ══════════════════════════════════════════════════════
CONFIG = {
    "data_folder":    "./data",
    "vector_db_path": "./chroma_rag_db",
    "collection":     "rag_complete",
    "embed_model":    "nomic-embed-text",   # Ollama embedding
    "llm_model":      "llama3.2",           # Ollama LLM
    "chunk_size":     1000,
    "chunk_overlap":  150,
    "top_k":          5,
    "rebuild_index":  False,   # True = re-embed everything
}


# ══════════════════════════════════════════════════════
# STAGE 1: INGESTION
# ══════════════════════════════════════════════════════
def stage1_ingest(folder: str) -> list[Document]:
    print(f"\n[Stage 1] Ingesting from: {folder}")
    from langchain_community.document_loaders import (
        PyMuPDFLoader, CSVLoader, TextLoader,
        BSHTMLLoader, UnstructuredMarkdownLoader, DirectoryLoader
    )
    import pandas as pd

    all_docs = []
    p = Path(folder)

    loaders = [
        ("*.pdf",   PyMuPDFLoader,              "pdf"),
        ("*.txt",   TextLoader,                 "txt"),
        ("*.md",    UnstructuredMarkdownLoader,  "markdown"),
        ("*.html",  BSHTMLLoader,               "html"),
    ]
    for glob, cls, dtype in loaders:
        for f in p.glob(glob):
            try:
                docs = cls(str(f)).load()
                for d in docs: d.metadata["data_type"] = dtype
                all_docs.extend(docs)
                print(f"  [{dtype}] {f.name} → {len(docs)} docs")
            except Exception as e:
                print(f"  ERROR {f.name}: {e}")

    for f in p.glob("*.csv"):
        try:
            df = pd.read_csv(f)
            for idx, row in df.iterrows():
                all_docs.append(Document(
                    page_content=row.to_string(),
                    metadata={"source": str(f), "data_type": "csv", "row": idx}
                ))
            print(f"  [csv] {f.name} → {len(df)} rows")
        except Exception as e:
            print(f"  ERROR {f.name}: {e}")

    print(f"[Stage 1] ✅ {len(all_docs)} documents ingested")
    return all_docs


# ══════════════════════════════════════════════════════
# STAGE 2: CHUNKING
# ══════════════════════════════════════════════════════
def stage2_chunk(docs: list[Document]) -> list[Document]:
    print(f"\n[Stage 2] Chunking {len(docs)} documents...")
    from langchain_text_splitters import (
        RecursiveCharacterTextSplitter, MarkdownHeaderTextSplitter
    )

    pdf_txt_splitter = RecursiveCharacterTextSplitter(
        chunk_size=CONFIG["chunk_size"],
        chunk_overlap=CONFIG["chunk_overlap"],
        separators=["\n\n", "\n", ". ", " "],
        add_start_index=True
    )
    md_header_splitter = MarkdownHeaderTextSplitter(
        headers_to_split_on=[("#","H1"),("##","H2"),("###","H3")]
    )
    md_sub = RecursiveCharacterTextSplitter(chunk_size=800, chunk_overlap=100)

    chunks, buckets = [], {"text":[], "md":[], "pass":[]}
    for d in docs:
        dt = d.metadata.get("data_type","")
        if dt in ("pdf","txt","html"): buckets["text"].append(d)
        elif dt == "markdown":           buckets["md"].append(d)
        else:                           buckets["pass"].append(d)

    chunks.extend(pdf_txt_splitter.split_documents(buckets["text"]))
    for d in buckets["md"]:
        splits = md_header_splitter.split_text(d.page_content)
        for s in splits: s.metadata.update(d.metadata)
        chunks.extend(md_sub.split_documents(splits))
    chunks.extend(buckets["pass"])

    chunks = [c for c in chunks if c.page_content.strip()]
    print(f"[Stage 2] ✅ {len(chunks)} chunks created")
    return chunks


# ══════════════════════════════════════════════════════
# STAGE 3 + 4: EMBED + VECTOR STORE
# ══════════════════════════════════════════════════════
def stage34_embed_and_store(chunks, rebuild=False):
    print(f"\n[Stage 3+4] Embedding + Vector Store...")
    from langchain_ollama import OllamaEmbeddings
    from langchain_chroma import Chroma

    embeddings = OllamaEmbeddings(model=CONFIG["embed_model"])
    db_path = CONFIG["vector_db_path"]
    col     = CONFIG["collection"]

    if rebuild or not Path(db_path).exists():
        print(f"  Building new index ({len(chunks)} chunks)...")
        vs = Chroma.from_documents(
            documents=chunks, embedding=embeddings,
            persist_directory=db_path, collection_name=col
        )
        print(f"[Stage 3+4] ✅ {vs._collection.count()} vectors stored")
    else:
        vs = Chroma(persist_directory=db_path,
                     embedding_function=embeddings, collection_name=col)
        print(f"[Stage 3+4] ✅ Loaded {vs._collection.count()} existing vectors")
    return vs


# ══════════════════════════════════════════════════════
# STAGE 5: RETRIEVER
# ══════════════════════════════════════════════════════
def stage5_retriever(vectorstore):
    print(f"\n[Stage 5] Building retriever (MMR, k={CONFIG['top_k']})...")
    retriever = vectorstore.as_retriever(
        search_type="mmr",
        search_kwargs={"k": CONFIG["top_k"], "fetch_k": CONFIG["top_k"] * 3}
    )
    print(f"[Stage 5] ✅ MMR retriever ready")
    return retriever


# ══════════════════════════════════════════════════════
# STAGE 6: LLM + RAG CHAIN
# ══════════════════════════════════════════════════════
def stage6_build_chain(retriever):
    print(f"\n[Stage 6] Loading LLM: {CONFIG['llm_model']}...")
    from langchain_ollama import ChatOllama

    llm = ChatOllama(model=CONFIG["llm_model"], temperature=0, num_ctx=4096)

    prompt = ChatPromptTemplate.from_messages([
        ("system", """You are a helpful AI assistant with access to a document database.
Answer questions using ONLY the provided context.
If the context doesn't contain the answer, say: "I don't have enough information in my documents to answer this."
Be accurate, concise, and cite the source when possible."""),
        ("human", "Context:\n{context}\n\nQuestion: {question}")
    ])

    def fmt(docs):
        return "\n\n---\n".join([
            f"[Source: {d.metadata.get('source','?')}]\n{d.page_content}"
            for d in docs
        ])

    chain = (
        RunnableParallel({"context": retriever | fmt, "question": RunnablePassthrough()})
        | prompt | llm | StrOutputParser()
    )
    print(f"[Stage 6] ✅ RAG chain ready")
    return chain, retriever


# ══════════════════════════════════════════════════════
# INTERACTIVE Q&A LOOP
# ══════════════════════════════════════════════════════
def interactive_qa(chain, retriever):
    print("\n" + "="*56)
    print("  RAG Pipeline Ready — Ask your documents anything!")
    print("  Type 'quit' to exit | 'sources' to see last sources")
    print("="*56 + "\n")

    last_docs = []
    while True:
        try:
            q = input("You: ").strip()
            if not q: continue
            if q.lower() == "quit": print("Goodbye!"); break
            if q.lower() == "sources":
                print("Sources from last query:")
                for d in last_docs:
                    print(f"  - {d.metadata.get('source','?')} (p.{d.metadata.get('page','-')})")
                continue

            last_docs = retriever.invoke(q)
            print(f"\nAssistant: ", end="", flush=True)
            for chunk in chain.stream(q):
                print(chunk, end="", flush=True)
            print(f"\n[Retrieved {len(last_docs)} chunks | type 'sources' to see them]\n")

        except KeyboardInterrupt:
            print("\nGoodbye!"); break


# ══════════════════════════════════════════════════════
# MAIN ENTRY POINT
# ══════════════════════════════════════════════════════
if __name__ == "__main__":
    rebuild = "--rebuild" in sys.argv
    CONFIG["rebuild_index"] = rebuild

    print("╔══════════════════════════════════════════╗")
